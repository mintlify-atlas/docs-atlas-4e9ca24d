## Overview

The `DashboardController` manages dashboard metadata, data retrieval, and layouts for Renovation Dashboard system.

## Dashboard Retrieval

### getDashboardMeta()

Retrieves metadata for a specific dashboard.

```typescript
const response = await core.dashboard.getDashboardMeta({
  dashboardName: 'Sales Balance'
});
```

      Name of the dashboard
    </ParamField>
  </Expandable>
</ParamField>

<ResponseField name="data" type="FrappeDashboard">
  Dashboard metadata
  <Expandable title="FrappeDashboard properties">
    <ResponseField name="name" type="string">

## Data Management

### refreshData()

Refreshes dashboard data from the backend.

```typescript
// Refresh all dashboards
await core.dashboard.refreshData();

// Refresh specific dashboard
await core.dashboard.refreshData({
  dashboardName: 'Sales Balance'
});
```

<ParamField path="refreshDataParams" type="RefreshDataParams">
  Refresh parameters
  <Expandable title="RefreshDataParams properties">
    <ParamField path="dashboardName" type="string">
      Specific dashboard to refresh. If not specified, refreshes all
    </ParamField>
  </Expandable>
</ParamField>

<ResponseField name="data" type="string | null">
  Returns null on success
</ResponseField>

### dashboardDataListener

Observable for listening to dashboard data changes.

```typescript
core.dashboard.dashboardDataListener.subscribe(response => {
  if (response && response.success) {
    console.log('Dashboard updated:', response.data);
  }
});
```

<ResponseField name="dashboardDataListener" type="BehaviorSubject<RequestResponse<any>>">
  Observable that emits when dashboard data updates
</ResponseField>

## Cache Management

### clearCache()

Clears dashboard metadata and data cache.

```typescript
core.dashboard.clearCache();
```

### dashboardMetaCache

Direct access to dashboard metadata cache.

```typescript
// Get cached dashboard
const cached = core.dashboard.dashboardMetaCache['Sales Balance'];

// Check if cached
if (core.dashboard.dashboardMetaCache['Sales Balance']) {
  console.log('Dashboard is cached');
}
```

<ResponseField name="dashboardMetaCache" type="{ [dashboardName: string]: FrappeDashboard }">
  Cached dashboard metadata
</ResponseField>

<Note>
Requires the `renovation_core` app to be installed.
</Note>

```typescript Load and Display Dashboard
  dashboardName: 'Sales Balance'

  }
```
```typescript
}
```

### Multiple Dashboards

```typescript
async function loadMultipleDashboards(dashboardNames: string[]) {
  const results = [];
  
  for (const name of dashboardNames) {
    // Get metadata
    const metaResponse = await core.dashboard.getDashboardMeta({
      dashboardName: name
    });
    
    if (metaResponse.success) {
      // Get data
      const dataResponse = await core.dashboard.getDashboardData({
        dashboardName: name
      });
      
      results.push({
        name,
        meta: metaResponse.data,
        data: dataResponse.success ? dataResponse.data : null
      });
    }
  }
  
  return results;
}

// Usage
const dashboards = await loadMultipleDashboards([
  'Sales Balance',
  'Revenue Overview',
  'Customer Stats'
]);

dashboards.forEach(dashboard => {
  console.log(`${dashboard.name}:`, dashboard.data);
});
```

### Parameter Validation

```typescript
async function validateAndGetDashboardData(
  dashboardName: string,
  userParams: { [key: string]: any }
) {
  // Get dashboard metadata
  const metaResponse = await core.dashboard.getDashboardMeta({
    dashboardName
  });
  
  if (!metaResponse.success) {
    throw new Error('Dashboard not found');
  }
  
  const dashboard = metaResponse.data;
  
  // Build parameters with validation
  const params = dashboard.params.map(param => {
    const value = userParams[param.param] || param.default_value;
    
    // Check required parameters
    if (param.reqd && !value) {
      throw new Error(`Required parameter missing: ${param.label}`);
    }
    
    return {
      ...param,
      value
    };
  });
  
  // Get data
  return await core.dashboard.getDashboardData({
    dashboardName,
    params
  });
}

// Usage
try {
  const response = await validateAndGetDashboardData('Sales Analytics', {
    from_date: '2024-01-01',
    to_date: '2024-12-31'
  });
  
  console.log('Dashboard data:', response.data);
} catch (error) {
  console.error('Error:', error.message);
}
```
