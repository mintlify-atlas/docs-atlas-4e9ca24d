## Overview

`DocField` is a class that contains metadata and properties of a document field in Frappe. It provides helper functions to parse field information from Frappe format and manage field definitions, types, and behavior.

## Class Structure

```typescript
class DocField {
  [x: string]: any;
  
  static fromFrappeDocField(field: FrappeDocField): DocField
}
```

## Static Methods

### fromFrappeDocField(field)

Parses a DocField from Frappe format into a structured `DocField` instance. This method performs transformations including:

<ParamField path="field" type="FrappeDocField" required>
  The raw Frappe field metadata object containing all field information.
</ParamField>

<ResponseField name="return" type="DocField">
  A fully parsed and structured `DocField` instance.
</ResponseField>

#### Transformations

1. **Boolean Conversion**: Converts Frappe's 0/1 values to true/false for boolean properties
2. **Naming Convention**: Transforms underscore_case property names to camelCase
3. **Property Mapping**: Maps all Frappe properties to the DocField instance

### Display Options

<ParamField path="inListView" type="boolean" default="false">
  Whether the field should appear as a column in the list view of the doctype.
</ParamField>

<ParamField path="inStandardFilter" type="boolean" default="false">

</ParamField>

### Field Configuration

<ParamField path="options" type="string" default="''">
  Field-type specific configuration:
  - **Link**: Contains the target DocType name
  - **Select**: Contains newline-separated list of options
  - **Dynamic Link**: Contains the field that holds the DocType name
  - **Table**: Contains the child DocType name
</ParamField>

## Usage Examples

### Parsing from Frappe Format

```typescript
import DocField from '@renovationjs/core/model/docfield';

// API response from Frappe
const frappeField = {
  fieldname: 'item_code',
  fieldtype: 'Data',
  label: 'Item Code',
  reqd: 1,
  in_list_view: 1,
  in_standard_filter: 1,
  read_only: 0,
  hidden: 0
};

// Parse to DocField
const field = DocField.fromFrappeDocField(frappeField);

console.log(field.fieldname); // 'item_code'
console.log(field.reqd); // true (converted from 1)
console.log(field.inListView); // true (from in_list_view: 1)
console.log(field.readOnly); // false (from read_only: 0)
```

### Working with Link Fields

```typescript
// Link field configuration
const linkField = DocField.fromFrappeDocField({
  fieldname: 'customer',
  fieldtype: 'Link',
  label: 'Customer',
  options: 'Customer', // Target DocType
  reqd: 1,
  in_list_view: 1
});

if (linkField.fieldtype === 'Link') {
  console.log(`Links to: ${linkField.options}`);
  // Output: Links to: Customer
}
```

### Working with Select Fields

```typescript
// Select field with options
const selectField = DocField.fromFrappeDocField({
  fieldname: 'status',
  fieldtype: 'Select',
  label: 'Status',
  options: 'Draft\nActive\nInactive\nArchived',
  reqd: 1
});

// Parse options
const options = selectField.options.split('\n');
console.log('Available options:', options);
// Output: ['Draft', 'Active', 'Inactive', 'Archived']

// Build select dropdown
function buildSelectOptions(field: DocField): string[] {
  if (field.fieldtype === 'Select' && field.options) {
    return field.options.split('\n').filter(opt => opt.trim());
  }
  return [];
}
```

### Building Form Inputs

```typescript
function generateFormInput(field: DocField) {
  const baseConfig = {
    name: field.fieldname,
    label: field.label,
    required: field.reqd,
    disabled: field.readOnly,
    hidden: field.hidden
  };

  switch (field.fieldtype) {
    case 'Data':
    case 'Text':
      return { ...baseConfig, type: 'text' };
    
    case 'Int':
    case 'Float':
    case 'Currency':
      return { ...baseConfig, type: 'number' };
    
    case 'Date':
      return { ...baseConfig, type: 'date' };
    
    case 'Check':
      return { ...baseConfig, type: 'checkbox' };
    
    case 'Select':
      return {
        ...baseConfig,
        type: 'select',
        options: field.options.split('\n')
      };
    
    case 'Link':
      return {
        ...baseConfig,
        type: 'autocomplete',
        doctype: field.options
      };
    
    default:
      return { ...baseConfig, type: 'text' };
  }
}

const inputConfig = generateFormInput(field);
```

### Validating Field Values

```typescript
function validateFieldValue(field: DocField, value: any): boolean {
  // Check if required field is empty
  if (field.reqd && !value) {
    console.error(`${field.label} is required`);
    return false;
  }

  // Type-specific validation
  switch (field.fieldtype) {
    case 'Int':
      if (!Number.isInteger(Number(value))) {
        console.error(`${field.label} must be an integer`);
        return false;
      }
      break;
    
    case 'Email':
      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      if (!emailRegex.test(value)) {
        console.error(`${field.label} must be a valid email`);
        return false;
      }
      break;
    
    case 'Select':
      const options = field.options.split('\n');
      if (!options.includes(value)) {
        console.error(`${field.label} must be one of: ${options.join(', ')}`);
        return false;
      }
      break;
  }

  return true;
}
```

### Creating Field Groups

```typescript
interface FieldGroup {
  label: string;
  fields: DocField[];
}

function organizeFieldsIntoGroups(fields: DocField[]): FieldGroup[] {
  const groups: FieldGroup[] = [];
  let currentGroup: FieldGroup = { label: 'Default', fields: [] };

  for (const field of fields) {
    if (field.fieldtype === 'Section Break') {
      // Start new section
      if (currentGroup.fields.length > 0) {
        groups.push(currentGroup);
      }
      currentGroup = {
        label: field.label || 'Section',
        fields: []
      };
    } else if (field.fieldtype !== 'Column Break') {
      // Add field to current section
      currentGroup.fields.push(field);
    }
  }

  if (currentGroup.fields.length > 0) {
    groups.push(currentGroup);
  }

  return groups;
}
```
