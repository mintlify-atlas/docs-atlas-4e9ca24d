## Overview

The `MetaController` handles operations related to doctype metadata, including retrieving doctype definitions, field information, document counts, and report metadata.

## Doctype Metadata

### getDocMeta()

Retrieves doctype metadata from cache or backend.

```typescript
const response = await core.meta.getDocMeta({
  doctype: 'Sales Invoice'
});
```

</ParamField>

<ResponseField name="data" type="DocType">
  Doctype metadata including fields, permissions, and settings
  <Expandable title="DocType properties">

## Cache Management

### clearCache()

Clears all cached doctype metadata.

```typescript
core.meta.clearCache();
```

This forces fresh retrieval of all doctype metadata on next access.

### docTypeCache

Direct access to the doctype cache object.

```typescript
// Get cached doctype
const cachedDocType = core.meta.docTypeCache['Sales Invoice'];

// Check if doctype is cached
if (core.meta.docTypeCache['Sales Invoice']) {
  console.log('DocType is cached');
}
```

<ResponseField name="docTypeCache" type="{ [doctype: string]: DocType }">
  Object containing cached doctype metadata
</ResponseField>

<Warning>
Direct cache manipulation is not recommended. Use `clearCache()` to invalidate cache.
</Warning>

## Code Examples

### Working with Doctype Metadata

<CodeGroup>
```typescript Get Field Information
const response = await core.meta.getDocMeta({
  doctype: 'Sales Invoice'
});

if (response.success) {
  const doctype = response.data;
  
  // Get all fields
  console.log('Fields:', doctype.fields.length);
  
  // Find specific field
  const customerField = doctype.fields.find(
    f => f.fieldname === 'customer'
  );
  
  console.log('Customer field type:', customerField.fieldtype);
  console.log('Is required:', customerField.reqd);
  
  // Get translated label
  const label = await core.meta.getFieldLabel({
    doctype: 'Sales Invoice',
    fieldname: 'customer'
  });
  console.log('Translated label:', label);
}
```

```typescript Document Statistics
// Get document count
const countResponse = await core.meta.getDocCount({
  doctype: 'Sales Invoice',
  filters: {
    status: 'Draft',
    posting_date: ['>', '2024-01-01']
  }
});

console.log(`Found ${countResponse.data} draft invoices`);

// Get document info
const infoResponse = await core.meta.getDocInfo({
  doctype: 'Sales Invoice',
  docname: 'SI-001'
});

if (infoResponse.success) {
  const info = infoResponse.data;
  console.log('Attachments:', info.attachments.length);
  console.log('Comments:', info.total_comments);
  console.log('Versions:', info.versions.length);
}
```

```typescript Report Metadata
const response = await core.meta.getReportMeta({
  report: 'Sales Analytics'
});

if (response.success) {
  const report = response.data;
  
  console.log('Report name:', report.name);
  console.log('Available filters:');
  
  report.filters.forEach(filter => {
    console.log(`  ${filter.label} (${filter.fieldtype})`);
    if (filter.default_value) {
      console.log(`    Default: ${filter.default_value}`);
    }
  });
}
```

### Dynamic Form Building

```typescript
// Build form based on metadata
const response = await core.meta.getDocMeta({
  doctype: 'Sales Invoice'
});

if (response.success) {
  const doctype = response.data;
  
  // Filter visible fields
  const visibleFields = doctype.fields.filter(
    f => !f.hidden && !f.is_custom
  );
  
  // Group by field type
  const fieldsByType = visibleFields.reduce((acc, field) => {
    acc[field.fieldtype] = acc[field.fieldtype] || [];
    acc[field.fieldtype].push(field);
    return acc;
  }, {});
  
  // Build form sections
  const sections = [];
  let currentSection = null;
  
  for (const field of visibleFields) {
    if (field.fieldtype === 'Section Break') {
      currentSection = {
        label: field.label,
        fields: []
      };
      sections.push(currentSection);
    } else if (currentSection) {
      currentSection.fields.push({
        name: field.fieldname,
        label: field.label,
        type: field.fieldtype,
        required: field.reqd,
        options: field.options
      });
    }
  }
  
  console.log('Form sections:', sections);
}
```

### Checking Field Properties

```typescript
async function isFieldEditable(doctype: string, fieldname: string): Promise<boolean> {
  const response = await core.meta.getDocMeta({ doctype });
  
  if (response.success) {
    const field = response.data.fields.find(f => f.fieldname === fieldname);
    
    if (!field) return false;
    
    return !field.read_only && 
           !field.hidden && 
           field.fieldtype !== 'Section Break' &&
           field.fieldtype !== 'Column Break';
  }
  
  return false;
}

// Usage
const editable = await isFieldEditable('Sales Invoice', 'customer');
console.log('Customer field editable:', editable);
```
