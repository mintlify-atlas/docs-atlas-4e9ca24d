## Overview

The `PermissionController` manages document and doctype permissions, checking user access rights for various operations.

## Permission Checking

### hasPerm()

Checks if the user has a specific permission for a doctype.

```typescript
const hasRead = await core.perm.hasPerm({
  doctype: 'Sales Invoice',
  ptype: 'read',
  permLevel: 0,
  docname: 'SI-001'
});
```

<ParamField path="hasPermParams" type="HasPermParams" required>
  Permission check parameters
  <Expandable title="HasPermParams properties">
    <ParamField path="doctype" type="string" required>
      Target doctype
    </ParamField>
    <ParamField path="ptype" type="PermissionType" required>
      Permission type: 'read' | 'write' | 'create' | 'delete' | 'submit' | 'cancel' | 'amend'
    </ParamField>
    <ParamField path="permLevel" type="number">
      Permission level. Default: 0
    </ParamField>
    <ParamField path="docname" type="string">
      Specific document name for document-level permissions
    </ParamField>
  </Expandable>
</ParamField>

<ResponseField name="return" type="boolean">
  True if user has the permission, false otherwise
</ResponseField>

### hasPerms()

Checks if the user has multiple permissions for a doctype.

```typescript
const hasAll = await core.perm.hasPerms({
  doctype: 'Sales Invoice',
  ptypes: ['read', 'write', 'submit'],
  docname: 'SI-001'
});
```

<ParamField path="hasPermsParams" type="HasPermsParams" required>
  Multiple permission check parameters
  <Expandable title="HasPermsParams properties">
    <ParamField path="doctype" type="string" required>
      Target doctype
    </ParamField>
    <ParamField path="ptypes" type="PermissionType[]" required>
      Array of permission types to check
    </ParamField>
    <ParamField path="docname" type="string">
      Specific document name
    </ParamField>
  </Expandable>
</ParamField>

<ResponseField name="return" type="boolean">
  True if user has ALL specified permissions, false otherwise
</ResponseField>


## Code Examples

### Permission-Based UI

<CodeGroup>
```typescript Check Before Action
// Check if user can create
const canCreate = await core.perm.canCreate({
  doctype: 'Sales Invoice'
});

if (canCreate) {
  // Show create button
  const doc = await core.model.newDoc({
    doctype: 'Sales Invoice'
  });
}

// Check multiple permissions
const hasFullAccess = await core.perm.hasPerms({
  doctype: 'Sales Invoice',
  ptypes: ['read', 'write', 'submit', 'cancel']
});

if (hasFullAccess) {
  console.log('User has full access');
}
```

```typescript Document-Level Permissions
// Check specific document permissions
const canEdit = await core.perm.hasPerm({
  doctype: 'Sales Invoice',
  ptype: 'write',
  docname: 'SI-001'
});

if (canEdit) {
  // Allow editing
} else {
  // Show read-only view
}
```

```typescript Conditional Feature Access
// Check various permissions
const permissions = {
  canCreate: await core.perm.canCreate({ doctype: 'Item' }),
  canImport: await core.perm.canImport({ doctype: 'Item' }),
  canExport: await core.perm.canExport({ doctype: 'Item' }),
  canDelete: await core.perm.canDelete({ doctype: 'Item' })
};

// Enable/disable UI features
if (permissions.canImport) {
  showImportButton();
}

if (permissions.canExport) {
  showExportButton();
}
```
</CodeGroup>

### Permission Validation

```typescript
async function validateDocumentAccess(
  doctype: string,
  action: 'read' | 'write' | 'delete'
): Promise<void> {
  const perm = await core.perm.hasPerm({
    doctype,
    ptype: action
  });
  
  if (!perm) {
    throw new Error(
      `You don't have ${action} permission for ${doctype}`
    );
  }
}

// Usage
try {
  await validateDocumentAccess('Sales Invoice', 'write');
  // Proceed with write operation
} catch (error) {
  console.error(error.message);
}
```
