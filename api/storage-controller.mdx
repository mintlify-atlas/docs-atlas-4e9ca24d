## Overview

The `StorageController` handles file uploads, folder management, and file URL generation with support for both Socket.IO and HTTP upload methods.

## File Upload

### uploadFile()

Uploads a file to the backend with progress tracking.

```typescript
const uploadStatus = core.storage.uploadFile({
  file: fileObject,
  isPrivate: false,
  folder: 'Home/Attachments',
  doctype: 'Sales Invoice',
  docname: 'SI-001',
  docfield: 'attachment'
});

uploadStatus.subscribe(status => {
  if (status.status === 'uploading') {
    console.log('Progress:', status.progress + '%');
  } else if (status.status === 'completed') {
    console.log('Upload complete:', status.requestResponse.data);
  }
});
```

<Note>
Automatically falls back to HTTP upload if Socket.IO upload fails.
</Note>

## Folder Management

### createFolder()

Creates a new folder in the file system.

```typescript
const response = await core.storage.createFolder(
  'Invoices',
  'Home/Documents'
);
```

<ParamField path="folderName" type="string" required>
  Name of the folder to create
</ParamField>

<ParamField path="parentFolder" type="string">
  Parent folder path. Default: 'Home'
</ParamField>

<ResponseField name="data" type="any">
  Created folder information
</ResponseField>

<Warning>
Folder name cannot contain forward slashes (/).
</Warning>

### checkFolderExists()

Checks if a folder exists.

```typescript
const response = await core.storage.checkFolderExists(
  'Home/Documents/Invoices'
);

if (response.data) {
  console.log('Folder exists');
}
```

<ParamField path="folderDir" type="string" required>
  Full folder path
</ParamField>

<ResponseField name="data" type="boolean">
  True if folder exists, false otherwise
</ResponseField>

## URL Generation

### getUrl()

Generates the complete URL to access a file or resource.

```typescript
const fileUrl = core.storage.getUrl('/files/invoice.pdf');
// Returns: 'https://example.com/files/invoice.pdf'

const imageUrl = core.storage.getUrl('/files/logo.png');
// Returns: 'https://example.com/files/logo.png'
```

<ParamField path="ref" type="string" required>
  Relative path or URL reference
</ParamField>

<ResponseField name="return" type="string">
  Complete URL with host
</ResponseField>

<Note>
Returns the input as-is if it's already a complete URL (starts with 'http').
</Note>

### File Upload Scenarios

<CodeGroup>
```typescript Browser File Upload
// From file input
const fileInput = document.querySelector('input[type="file"]');
const file = fileInput.files[0];

const uploadStatus = core.storage.uploadFile({
  file: file,
  isPrivate: false,

      break;
  }
```

### Progress Tracking

```typescript
let uploadSubscription;

function uploadWithProgress(file: File) {
  const uploadStatus = core.storage.uploadFile({
    file: file,
    folder: 'Home/Uploads'
  });
  
  uploadSubscription = uploadStatus.subscribe(status => {
    if (status.status === 'uploading' && status.hasProgress) {
      // Update UI progress bar
      const progressBar = document.querySelector('.progress-bar');
      progressBar.style.width = `${status.progress}%`;
      progressBar.textContent = `${status.progress}%`;
    } else if (status.status === 'completed') {
      const response = status.requestResponse;
      if (response.success) {
        showSuccess('File uploaded successfully!');
        console.log('File URL:', response.data.file_url);
      }
      uploadSubscription.unsubscribe();
    } else if (status.status === 'error') {
      showError('Upload failed: ' + status.requestResponse.error.title);
      uploadSubscription.unsubscribe();
    }
  });
}
```

### Folder Management

```typescript
async function ensureFolderExists(folderPath: string): Promise<boolean> {
  // Check if folder exists
  const existsResponse = await core.storage.checkFolderExists(folderPath);
  
  if (existsResponse.data) {
    return true;
  }
  
  // Create folder
  const parts = folderPath.split('/');
  const folderName = parts.pop();
  const parentFolder = parts.join('/') || 'Home';
  
  const createResponse = await core.storage.createFolder(
    folderName,
    parentFolder
  );
  
  return createResponse.success;
}

// Usage
const exists = await ensureFolderExists('Home/Documents/2024');
if (exists) {
  // Proceed with upload
  const uploadStatus = core.storage.uploadFile({
    file: file,
    folder: 'Home/Documents/2024'
  });
}
```

### Multiple File Upload

```typescript
async function uploadMultipleFiles(files: FileList) {
  const uploads = [];
  
  for (let i = 0; i < files.length; i++) {
    const file = files[i];
    
    const uploadPromise = new Promise((resolve, reject) => {
      const uploadStatus = core.storage.uploadFile({
        file: file,
        folder: 'Home/Uploads'
      });
      
      uploadStatus.subscribe(status => {
        if (status.status === 'completed') {
          if (status.requestResponse.success) {
            resolve(status.requestResponse.data);
          } else {
            reject(status.requestResponse.error);
          }
        } else if (status.status === 'error') {
          reject(status.requestResponse.error);
        }
      });
    });
    
    uploads.push(uploadPromise);
  }
  
  try {
    const results = await Promise.all(uploads);
    console.log('All files uploaded:', results);
    return results;
  } catch (error) {
    console.error('Upload failed:', error);
    throw error;
  }
}
```

### Working with URLs

```typescript
// Get full URLs for file references
const invoice = await core.model.getDoc({
  doctype: 'Sales Invoice',
  docname: 'SI-001'
});

if (invoice.success && invoice.data.attachment) {
  // Convert relative path to full URL
  const fullUrl = core.storage.getUrl(invoice.data.attachment);
  console.log('Attachment URL:', fullUrl);
  
  // Can be used directly in <img> or <a> tags
  // <img src={fullUrl} />
  // <a href={fullUrl}>Download</a>
}

// Works with already complete URLs
const externalUrl = core.storage.getUrl('https://example.com/file.pdf');
// Returns: 'https://example.com/file.pdf' (unchanged)
```
